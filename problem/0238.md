```c++
class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        int n = nums.size();
        vector<int> r(n);
        int rp = 1;
        for (int i = n - 1; i >= 0; i--) {
            r[i] = rp;
            rp *= nums[i];
        }
        int lp = 1;
        for (int i = 0; i < n; i++) {
            r[i] *= lp;
            lp *= nums[i];
        }
        return r;
    }
};
```
