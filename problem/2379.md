```c++
class Solution {
public:
    int minimumRecolors(string blocks, int k) {
        int whiteCount = 0;
        int minC = k;
        int r = 0;
        for (int r = 0; r < blocks.size(); r++) {
            if (blocks[r] == 'W') whiteCount++;
            if (r == k - 1) {
                minC = whiteCount;
            } else if (r >= k) {
                if (blocks[r - k] == 'W') whiteCount--;
                minC = min(minC, whiteCount);
            }
            if (minC == 0) return 0;
        }
        return minC;
    }
};
```
