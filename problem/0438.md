```c++
class Solution {
public:
    vector<int> findAnagrams(string s, string p) {
        vector<int> ep(26, 0);
        vector<int> mp(26, 0);
        for (auto c : p) {
            mp[c - 'a']++;
        }
        int n = p.size();
        vector<int> res;
        for (int l = 0, r = 0; r < s.size(); r++) {
            mp[s[r] - 'a']--;
            if (r - l + 1 == n) {
                if (mp == ep) res.push_back(l);
                mp[s[l++] - 'a']++;
            }
        }
        return res;
    }
};
```
